sfr （特殊功能寄存器声明）

例如：sfr P0 = 0x80;

例如：sbit P0_1 = 0x81；或者 sbit P0_1 = P0^1;

# 74HC595(移位寄存器)工作原理

1. 74HC595有三个引脚，分别是P3^4,P3^5,P3^6**

2. 当P3^6为高电平的时候（默认为低电平），P3^6引脚会读取P3^4引脚的状态。
   当P3^4引脚为高电平，则写入1，如果是低电平，则写入0（从高位往低位写入，需要手动移位）

3. 当写入了八个位的数据之后，只需要给P3^5一个高电平。当它得到信号之后，就会将写入的八个数一次性赋给八个引脚，并将数据全部传出去

4. SER输入；SRCLK确认；RCLK（打包转移）
   **P3^4数据位，P3^6数据移位位,P3^5并行输出位**
   
   ---
   
   ## 使用样例（通俗易懂）

```
#include "REGX52.H"  // 替换成REGX52.H，引脚定义逻辑不变

/*************************
  引脚定义（适配REGX52.H）
  SER：P3^4  串行数据输入（输入）
  SRCLK：P3^6 移位时钟（确认）
  RCLK：P3^5  锁存时钟（打包转移）
*************************/
sbit SER   = P3^4;  // REGX52.H直接支持P3^4格式
sbit SRCLK = P3^6;
sbit RCLK  = P3^5;

/*************************
  延时函数：简单易懂，控制亮灭速度
*************************/
void Delay(unsigned int t)
{
    unsigned int i, j;
    for(i=t; i>0; i--)
        for(j=110; j>0; j--);
}

/*************************
  74HC595发送1字节数据
  逻辑：先移位（打包），后锁存（发货）
*************************/
void HC595_SendByte(unsigned char dat)
{
    unsigned char i;
    // 1. 逐位移入8位数据（高位先行）
    for(i=0; i<8; i++)
    {
        SER = dat >> 7;    // 取当前最高位传给SER
        dat <<= 1;         // 数据左移，准备下一位
        // 移位时钟上升沿触发（REGX52.H直接操作引脚电平）
        SRCLK = 0;
        SRCLK = 1;
        SRCLK = 0;
    }
    // 2. 锁存时钟上升沿，输出8位数据
    RCLK = 0;
    RCLK = 1;
    RCLK = 0;
}

/*************************
  主函数：驱动8个LED轮流亮
*************************/
void main()
{
    unsigned char led_dat = 0x80;  // 初始让第1个LED亮
    // 初始化引脚电平（REGX52.H直接赋值）
    SRCLK = 0;
    RCLK  = 0;

    while(1)
    {
        HC595_SendByte(led_dat);  // 发送数据到74HC595
        Delay(500);               // 延时500ms，看清效果

        // 数据右移，实现LED轮流亮
        if(led_dat == 0x01)
            led_dat = 0x80;
        else
            led_dat >>= 1;
    }
}
```
